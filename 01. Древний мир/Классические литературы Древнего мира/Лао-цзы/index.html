    –Ф–∞–Њ, –Ї–Њ—В–Њ—А–Њ–µ –Љ–Њ–ґ–µ—В –±—Л—В—М –≤—Л—А–∞–ґ–µ–љ–Њ —Б–ї–Њ–≤–∞–Љ–Є, –љ–µ –µ—Б—В—М –≤–µ—З–љ–Њ–µ –Ф–∞–Њ. –Ш–Љ—П, –Ї–Њ—В–Њ—А–Њ–µ –Љ–Њ–ґ–µ—В –±—Л—В—М –љ–∞–Ј–≤–∞–љ–Њ, –љ–µ –µ—Б—В—М –≤–µ—З–љ–Њ–µ –Є–Љ—П. –С–µ–Ј—Л–Љ—П–љ–љ–Њ–µ –µ—Б—В—М –љ–∞—З–∞–ї–Њ –Э–µ–±–∞ –Є –Ч–µ–Љ–ї–Є. –Ш–Љ–µ—О—Й–µ–µ –Є–Љ—П –µ—Б—В—М –Љ–∞—В—М –≤—Б–µ—Е –≤–µ—Й–µ–є. –Я–Њ—Н—В–Њ–Љ—Г, –Ї—В–Њ —Б–≤–Њ–±–Њ–і–µ–љ –Њ—В —Б—В—А–∞—Б—В–µ–є, –≤–Є–і–Є—В —З—Г–і–µ—Б–љ—Г—О —В–∞–є–љ—Г [–Ф–∞–Њ], –∞ –Ї—В–Њ –Є–Љ–µ–µ—В —Б—В—А–∞—Б—В–Є, –≤–Є–і–Є—В –µ–≥–Њ –≤ –Ї–Њ–љ–µ—З–љ–Њ–є —Д–Њ—А–Љ–µ. –≠—В–Є –і–≤–µ –≤–µ—Й–Є –Њ–і–љ–Њ–≥–Њ –њ—А–Њ–Є—Б—Е–Њ–ґ–і–µ–љ–Є—П, –љ–Њ —А–∞–Ј–љ—Л–µ –њ–Њ –љ–∞–Ј–≤–∞–љ–Є—О. –Ю–±–µ –Њ–љ–Є –љ–∞–Ј—Л–≤–∞—О—В—Б—П –≥–ї—Г–±–Њ–Ї–Є–Љ–Є. –У–ї—Г–±–Њ–Ї–Њ–µ –Є –µ—Й–µ –±–Њ–ї–µ–µ –≥–ї—Г–±–Њ–Ї–Њ–µ вАФ –≤—А–∞—В–∞ –≤—Б–µ—Е —З—Г–і–µ—Б.

    –У–ї–∞–≤–∞ 2¶

    –Ъ–Њ–≥–і–∞ –≤—Б–µ –≤ –Љ–Є—А–µ —Г–Ј–љ–∞—О—В, —З—В–Њ –њ—А–µ–Ї—А–∞—Б–љ–Њ–µ –µ—Б—В—М –њ—А–µ–Ї—А–∞—Б–љ–Њ–µ, –њ–Њ—П–≤–ї—П–µ—В—Б—П –Є –±–µ–Ј–Њ–±—А–∞–Ј–љ–Њ–µ. –Ъ–Њ–≥–і–∞ –≤—Б–µ —Г–Ј–љ–∞—О—В, —З—В–Њ –і–Њ–±—А–Њ–µ –µ—Б—В—М –і–Њ–±—А–Њ–µ, –њ–Њ—П–≤–ї—П–µ—В—Б—П –Є –Ј–ї–Њ. –Я–Њ—Н—В–Њ–Љ—Г –±—Л—В–Є–µ –Є –љ–µ–±—Л—В–Є–µ –њ–Њ—А–Њ–ґ–і–∞—О—В –і—А—Г–≥ –і—А—Г–≥–∞, —В—А—Г–і–љ–Њ–µ –Є –ї–µ–≥–Ї–Њ–µ —Б–Њ–Ј–і–∞—О—В –і—А—Г–≥ –і—А—Г–≥–∞, –і–ї–Є–љ–љ–Њ–µ –Є –Ї–Њ—А–Њ—В–Ї–Њ–µ —Б–Њ–Њ—В–љ–Њ—Б—П—В—Б—П –і—А—Г–≥ —Б –і—А—Г–≥–Њ–Љ, –≤—Л—Б–Њ–Ї–Њ–µ –Є –љ–Є–Ј–Ї–Њ–µ —Б—В—А–µ–Љ—П—В—Б—П –і—А—Г–≥ –Ї –і—А—Г–≥—Г, –≥–Њ–ї–Њ—Б –Є –Ј–≤—Г–Ї –≥–∞—А–Љ–Њ–љ–Є—А—Г—О—В –і—А—Г–≥ —Б –і—А—Г–≥–Њ–Љ, –њ—А–µ–і—Л–і—Г—Й–µ–µ –Є –њ–Њ—Б–ї–µ–і—Г—О—Й–µ–µ —Б–ї–µ–і—Г—О—В –і—А—Г–≥ –Ј–∞ –і—А—Г–≥–Њ–Љ. –Я–Њ—Н—В–Њ–Љ—Г —Б–Њ–≤–µ—А—И–µ–љ–љ–Њ–Љ—Г–і—А—Л–є —Б–Њ–≤–µ—А—И–∞–µ—В –љ–µ–і–µ—П–љ–Є–µ –Є –њ—А–µ–±—Л–≤–∞–µ—В –≤ –±–µ–Ј–Љ–Њ–ї–≤–љ–Њ–Љ —Г—З–µ–љ–Є–Є; –≤—Б–µ –≤–µ—Й–Є –≤–Њ–Ј–љ–Є–Ї–∞—О—В, –∞ –Њ–љ –љ–µ –Њ—В–Ї–∞–Ј—Л–≤–∞–µ—В—Б—П –Њ—В –љ–Є—Е; –Њ–љ –њ—А–Њ–Є–Ј–≤–Њ–і–Є—В, –љ–Њ –љ–µ –≤–ї–∞–і–µ–µ—В; –Њ–љ –і–µ–є—Б—В–≤—Г–µ—В, –љ–Њ –љ–µ –њ–Њ–ї–∞–≥–∞–µ—В—Б—П –љ–∞ —Б–≤–Њ–Є —Б–Є–ї—Л; –Њ–љ —Б–Њ–≤–µ—А—И–∞–µ—В –њ–Њ–і–≤–Є–≥–Є, –љ–Њ –љ–µ –њ—А–Є—Б–≤–∞–Є–≤–∞–µ—В –Є—Е —Б–µ–±–µ. –Ґ–Њ–ї—М–Ї–Њ –љ–µ –њ—А–Є—Б–≤–∞–Є–≤–∞—П, –Њ–љ —Б–Њ—Е—А–∞–љ—П–µ—В –Є—Е –љ–∞–≤—Б–µ–≥–і–∞.

    –У–ї–∞–≤–∞ 3¶

    –Э–µ –њ–Њ—З–Є—В–∞—П –і–Њ—Б—В–Њ–є–љ—Л—Е, –і–µ–ї–∞—О—В –љ–∞—А–Њ–і —Б–≤–Њ–±–Њ–і–љ—Л–Љ –Њ—В —Б–Њ–њ–µ—А–љ–Є—З–µ—Б—В–≤–∞. –Э–µ –і–Њ—А–Њ–ґ–∞ —А–µ–і–Ї–Є–Љ–Є –≤–µ—Й–∞–Љ–Є, –і–µ–ї–∞—О—В –љ–∞—А–Њ–і —Б–≤–Њ–±–Њ–і–љ—Л–Љ –Њ—В –≤–Њ—А–Њ–≤—Б—В–≤–∞. –Э–µ –њ–Њ–Ї–∞–Ј—Л–≤–∞—П —В–Њ–≥–Њ, —З—В–Њ –ґ–µ–ї–∞–љ–љ–Њ, –і–µ–ї–∞—О—В –љ–∞—А–Њ–і —Б–≤–Њ–±–Њ–і–љ—Л–Љ –Њ—В —Б–Љ—Г—В—Л. –Я–Њ—Н—В–Њ–Љ—Г –Љ—Г–і—А–µ—Ж, —Г–њ—А–∞–≤–ї—П—П –ї—О–і—М–Љ–Є, –Њ–њ—Г—Б—В–Њ—И–∞–µ—В –Є—Е —Б–µ—А–і—Ж–∞, –љ–Њ –љ–∞–њ–Њ–ї–љ—П–µ—В –Є—Е —З—А–µ–≤–∞, –Њ—Б–ї–∞–±–ї—П–µ—В –Є—Е –≤–Њ–ї—О, –љ–Њ —Г–Ї—А–µ–њ–ї—П–µ—В –Є—Е –Ї–Њ—Б—В–Є. –Я–Њ—Б—В–Њ—П–љ–љ–Њ –і–µ–ї–∞–µ—В –љ–∞—А–Њ–і –љ–µ–Ј–љ–∞—О—Й–Є–Љ –Є –±–µ—Б—Б—В—А–∞—Б—В–љ—Л–Љ, –∞ –Ј–љ–∞—О—Й–Є—Е вАФ –љ–µ —Б–Љ–µ—О—Й–Є–Љ–Є –і–µ–є—Б—В–≤–Њ–≤–∞—В—М. –Ю–љ —В–≤–Њ—А–Є—В –љ–µ–і–µ—П–љ–Є–µ, –Є –њ–Њ—В–Њ–Љ—Г –љ–µ—В –љ–Є—З–µ–≥–Њ, —З—В–Њ –љ–µ –±—Л–ї–Њ –±—Л —Г—Б—В—А–Њ–µ–љ–Њ.

    –У–ї–∞–≤–∞ 4¶

    –Ф–∞–Њ –њ—Г—Б—В–Њ, –љ–Њ –Ї–Њ–≥–і–∞ –Є–Љ –њ–Њ–ї—М–Ј—Г—О—В—Б—П, –Њ–љ–Њ –љ–µ –њ–µ—А–µ–њ–Њ–ї–љ—П–µ—В—Б—П. –У–ї—Г–±–Њ–Ї–Њ–µ, –Њ–љ–Њ –њ–Њ–і–Њ–±–љ–Њ –њ—А–∞—А–Њ–і–Є—В–µ–ї—О –≤—Б–µ—Е –≤–µ—Й–µ–є. –Ю–љ–Њ –њ—А–Є—В—Г–њ–ї—П–µ—В –Њ—Б—В—А–Њ—В—Г, —А–∞—Б–њ—Г—В—Л–≤–∞–µ—В —Г–Ј–ї—Л, —Б–Љ—П–≥—З–∞–µ—В —Б–≤–µ—В, —Б–Њ–µ–і–Є–љ—П–µ—В—Б—П —Б –њ—Л–ї—М—О. –Ю–љ–Њ –≥–ї—Г–±–Њ–Ї–Њ –Є –Ї–∞–ґ–µ—В—Б—П —Б—Г—Й–µ—Б—В–≤—Г—О—Й–Є–Љ –≤–µ—З–љ–Њ. –ѓ –љ–µ –Ј–љ–∞—О, —З–µ–є –Њ–љ–Њ —Б—Л–љ, вАФ –Њ–љ–Њ –њ—А–µ–і—И–µ—Б—В–≤—Г–µ—В –Т–ї–∞–і—Л–Ї–µ.

    –У–ї–∞–≤–∞ 5¶

    –Э–µ–±–Њ –Є –Ч–µ–Љ–ї—П –љ–µ –Њ–±–ї–∞–і–∞—О—В —З–µ–ї–Њ–≤–µ—З–љ–Њ—Б—В—М—О вАФ –і–ї—П –љ–Є—Е –≤—Б–µ –≤–µ—Й–Є –њ–Њ–і–Њ–±–љ—Л —Б–Њ–ї–Њ–Љ–µ–љ–љ—Л–Љ —Б–Њ–±–∞–Ї–∞–Љ. –°–Њ–≤–µ—А—И–µ–љ–љ–Њ–Љ—Г–і—А—Л–є –љ–µ –Њ–±–ї–∞–і–∞–µ—В —З–µ–ї–Њ–≤–µ—З–љ–Њ—Б—В—М—О вАФ –і–ї—П –љ–µ–≥–Њ –љ–∞—А–Њ–і –њ–Њ–і–Њ–±–µ–љ —Б–Њ–ї–Њ–Љ–µ–љ–љ—Л–Љ —Б–Њ–±–∞–Ї–∞–Љ. –Я—А–Њ—Б—В—А–∞–љ—Б—В–≤–Њ –Љ–µ–ґ–і—Г –Э–µ–±–Њ–Љ –Є –Ч–µ–Љ–ї–µ–є вАФ —А–∞–Ј–≤–µ –Њ–љ–Њ –љ–µ –њ–Њ–і–Њ–±–љ–Њ –Ї—Г–Ј–љ–µ—З–љ–Њ–Љ—Г –Љ–µ—Е—Г? –Я—Г—Б—В–Њ–µ, –љ–Њ –љ–µ–Є—Б—В–Њ—Й–Є–Љ–Њ–µ; —З–µ–Љ –±–Њ–ї—М—И–µ –њ—А–Є–≤–Њ–і–Є—И—М –µ–≥–Њ –≤ –і–≤–Є–ґ–µ–љ–Є–µ, —В–µ–Љ –±–Њ–ї—М—И–µ –Њ–љ–Њ –њ—А–Њ–Є–Ј–≤–Њ–і–Є—В. –Ь–љ–Њ–≥–Њ—Б–ї–Њ–≤–Є–µ –≤–µ–і–µ—В –Ї –Є—Б—В–Њ—Й–µ–љ–Є—О. –Ы—Г—З—И–µ –і–µ—А–ґ–∞—В—М—Б—П —Б–µ—А–µ–і–Є–љ—Л.

    –У–ї–∞–≤–∞ 6¶

    –Ф—Г—Е –і–Њ–ї–Є–љ—Л –±–µ—Б—Б–Љ–µ—А—В–µ–љ вАФ —Н—В–Њ –љ–∞–Ј—Л–≤–∞—О—В –≥–ї—Г–±–Њ–Ї–Є–Љ –ґ–µ–љ—Б–Ї–Є–Љ –љ–∞—З–∞–ї–Њ–Љ. –Т—А–∞—В–∞ –≥–ї—Г–±–Њ–Ї–Њ–≥–Њ –ґ–µ–љ—Б–Ї–Њ–≥–Њ –љ–∞—З–∞–ї–∞ вАФ —Н—В–Њ –Ї–Њ—А–µ–љ—М –Э–µ–±–∞ –Є –Ч–µ–Љ–ї–Є. –Ю–љ–Њ —Б—Г—Й–µ—Б—В–≤—Г–µ—В –≤–µ—З–љ–Њ, –Ї–∞–Ї –љ–Є—В—М, –Є –і–µ–є—Б—В–≤—Г–µ—В –±–µ–Ј —Г—Б–Є–ї–Є–є.

    –У–ї–∞–≤–∞ 7¶

    –Э–µ–±–Њ –Є –Ч–µ–Љ–ї—П –≤–µ—З–љ—Л. –Э–µ–±–Њ –Є –Ч–µ–Љ–ї—П –≤–µ—З–љ—Л –њ–Њ—В–Њ–Љ—Г, —З—В–Њ –Њ–љ–Є —Б—Г—Й–µ—Б—В–≤—Г—О—В –љ–µ –і–ї—П —Б–µ–±—П. –Я–Њ—Н—В–Њ–Љ—Г –Њ–љ–Є –Љ–Њ–≥—Г—В —Б—Г—Й–µ—Б—В–≤–Њ–≤–∞—В—М –≤–µ—З–љ–Њ. –Я–Њ—Н—В–Њ–Љ—Г —Б–Њ–≤–µ—А—И–µ–љ–љ–Њ–Љ—Г–і—А—Л–є —Б—В–∞–≤–Є—В —Б–µ–±—П –њ–Њ–Ј–∞–і–Є, –љ–Њ –Њ–Ї–∞–Ј—Л–≤–∞–µ—В—Б—П –≤–њ–µ—А–µ–і–Є; –Њ–љ –њ—А–µ–љ–µ–±—А–µ–≥–∞–µ—В —Б–Њ–±–Њ–є, –љ–Њ —Б–Њ—Е—А–∞–љ—П–µ—В —Б–µ–±—П. –Э–µ –њ–Њ—В–Њ–Љ—Г –ї–Є, —З—В–Њ –Њ–љ –±–µ—Б–Ї–Њ—А—Л—Б—В–µ–љ? –Ю–љ –Љ–Њ–ґ–µ—В –Њ—Б—Г—Й–µ—Б—В–≤–Є—В—М —Б–≤–Њ–µ ¬Ђ—П¬ї.

    –У–ї–∞–≤–∞ 8¶

    –Ф–Њ—Б—В–Є–≥–љ–Є –Ї—А–∞–є–љ–µ–є –њ—Г—Б—В–Њ—В—Л; —Е—А–∞–љ–Є –Є—Б–Ї—А–µ–љ–љ—О—О —В–Є—И–Є–љ—Г. –Т–µ—Й–Є –≤–Њ–Ј–љ–Є–Ї–∞—О—В –Є [—Б–љ–Њ–≤–∞] –Є—Б—З–µ–Ј–∞—О—В; —П –≤–Є–ґ—Г –Є—Е –≤–Њ–Ј–≤—А–∞—В. –Т–µ—Й–µ–є –Љ–љ–Њ–≥–Њ вАФ –Ї–∞–ґ–і–∞—П –≤–Њ–Ј–≤—А–∞—Й–∞–µ—В—Б—П –Ї —Б–≤–Њ–µ–Љ—Г –Ї–Њ—А–љ—О. –Т–Њ–Ј–≤—А–∞—В –Ї –Ї–Њ—А–љ—О –љ–∞–Ј—Л–≤–∞–µ—В—Б—П –њ–Њ–Ї–Њ–µ–Љ; –њ–Њ–Ї–Њ–є –љ–∞–Ј—Л–≤–∞–µ—В—Б—П –≤–Њ–Ј–≤—А–∞—В–Њ–Љ –Ї —Б—Г–і—М–±–µ; –≤–Њ–Ј–≤—А–∞—В –Ї —Б—Г–і—М–±–µ –љ–∞–Ј—Л–≤–∞–µ—В—Б—П –њ–Њ—Б—В–Њ—П–љ—Б—В–≤–Њ–Љ; –Ј–љ–∞–љ–Є–µ –њ–Њ—Б—В–Њ—П–љ—Б—В–≤–∞ –љ–∞–Ј—Л–≤–∞–µ—В—Б—П –њ—А–Њ—Б–≤–µ—В–ї–µ–љ–Є–µ–Љ. –Э–µ –Ј–љ–∞—П –њ–Њ—Б—В–Њ—П–љ—Б—В–≤–∞, –і–µ–є—Б—В–≤—Г–µ—И—М –љ–∞—Г–≥–∞–і вАФ –Є –љ–µ—Б–µ—И—М –±–µ–і—Г. –Ч–љ–∞—О—Й–Є–є –њ–Њ—Б—В–Њ—П–љ—Б—В–≤–Њ —Б–њ–Њ—Б–Њ–±–µ–љ –Ї –≤–Њ—Б–њ—А–Є—П—В–Є—О; —Б–њ–Њ—Б–Њ–±–љ—Л–є –Ї –≤–Њ—Б–њ—А–Є—П—В–Є—О вАФ —Б–њ—А–∞–≤–µ–і–ї–Є–≤; —Б–њ—А–∞–≤–µ–і–ї–Є–≤—Л–є вАФ –≤—Б–µ–Њ–±—К–µ–Љ–ї—О—Й; –≤—Б–µ–Њ–±—К–µ–Љ–ї—О—Й–Є–є вАФ –Њ–±–ї–∞–і–∞–µ—В –њ—А–Є—А–Њ–і–Њ–є –Э–µ–±–∞; –Њ–±–ї–∞–і–∞—О—Й–Є–є –њ—А–Є—А–Њ–і–Њ–є –Э–µ–±–∞ вАФ –Њ–±–ї–∞–і–∞–µ—В –Ф–∞–Њ; –Њ–±–ї–∞–і–∞—О—Й–Є–є –Ф–∞–Њ вАФ –і–Њ–ї–≥–Њ–≤–µ—З–µ–љ; –і–Њ –Ї–Њ–љ—Ж–∞ –ґ–Є–Ј–љ–Є –Њ–љ –љ–µ –њ–Њ–і–≤–µ—А–≥–∞–µ—В—Б—П –Њ–њ–∞—Б–љ–Њ—Б—В–Є.

    –У–ї–∞–≤–∞ 17¶

    –Т–µ–ї–Є—З–∞–є—И–Є–є [–њ—А–∞–≤–Є—В–µ–ї—М] вАФ —В–Њ—В, –Њ –Ї–Њ–Љ [–љ–∞—А–Њ–і] —В–Њ–ї—М–Ї–Њ –Ј–љ–∞–µ—В, —З—В–Њ –Њ–љ —Б—Г—Й–µ—Б—В–≤—Г–µ—В.
